### 递归实现

'''

    class Solution:
        def preorderTraversal(self, root: TreeNode) -> List[int]:
            res=[]
            self.preOrder(root,res)
    
            return res
        #定义递归函数
        def preOrder(self,root,res):
            #判断root是否为空
            if not root:
                return res
            res.append(root.val)
            self.preOrder(root.left,res)
            self.preOrder(root.right,res)
    
            return res


'''

### 栈实现

'''

    class Solution:
        def postorderTraversal(self, root: TreeNode) -> List[int]:
            stack, node, res = [], root, []
            while stack or node:
                while node:
                    stack.append(node)
                    res.append(node.val)
                    node = node.right 
                node = stack.pop()
                node = node.left  
            return reversed(res)

'''
